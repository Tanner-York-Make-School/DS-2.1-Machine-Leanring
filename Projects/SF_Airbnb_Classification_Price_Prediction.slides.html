Creating Classes¶

In [4]:
def price_class(val):
    """Determins what class a listing should be given"""
    if val <= 120:
        return 0
    elif val > 120 and val <= 240:
        return 1
    elif val > 240:
        return 2


sf['price'] = sf['price'].apply(lambda x: price_class(x))
